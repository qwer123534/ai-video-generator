<script>
    const prompt = document.getElementById('prompt');
    const model = document.getElementById('model');
    const generateBtn = document.getElementById('generate');
    const status = document.getElementById('status');
    const videoContainer = document.getElementById('videoContainer');
    const video = document.getElementById('video');
    const historyList = document.getElementById('historyList');

    const demoVideos = [
        "https://www.w3school.com.cn/i/movie.mp4",
        "https://sample-videos.com/video123/mp4/720/big_buck_bunny_720p_1mb.mp4"
    ];

    let history = JSON.parse(localStorage.getItem('videoHistory')) || [];

    function renderHistory() {
        historyList.innerHTML = '';
        if (history.length === 0) {
            historyList.innerHTML = '<div class="empty">暂无生成记录</div>';
            return;
        }
        history.forEach((item, i) => {
            const div = document.createElement('div');
            div.className = 'history-item';
            div.innerHTML = `
                <div>${item.prompt}</div>
                <small style="color:#666">模型：${item.modelName} · ${item.time}</small>
            `;
            div.onclick = () => {
                video.src = item.url;
                videoContainer.style.display = 'block';
            };
            historyList.appendChild(div);
        });
    }

    renderHistory();

    generateBtn.onclick = async () => {
        const text = prompt.value.trim();
        if (!text) {
            alert('请输入视频描述内容！');
            return;
        }

        generateBtn.disabled = true;
        status.className = 'loading';
        status.textContent = '⏳ AI正在生成视频，请稍候...';
        videoContainer.style.display = 'none';

        setTimeout(() => {
            const videoUrl = demoVideos[Math.floor(Math.random() * demoVideos.length)];
            video.src = videoUrl;
            videoContainer.style.display = 'block';
            status.className = 'success';
            status.textContent = '✅ 视频生成完成！';

            const record = {
                prompt: text,
                modelName: model.value === 'doubao' ? '豆包' : '即梦',
                url: videoUrl,
                time: new Date().toLocaleString()
            };
            history.unshift(record);
            if (history.length > 15) history.pop();
            localStorage.setItem('videoHistory', JSON.stringify(history));
            renderHistory();

            generateBtn.disabled = false;
        }, 3000);
    };
</script>
